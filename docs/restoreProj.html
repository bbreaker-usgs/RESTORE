Computation of Streamflow Characteristics for RESTORE Project

Overview

This script was developed to generate streamflow statistics for U.S. Geological Survey (USGS) streamflow data as a part of a RESTORE Act project. The focus of the project is to evaluate streamflow alteration and analyze long-term trends in streams that contribute flow to the Gulf of Mexico.

Software

If R (version 3.3 or greater) is not already installed on your computer, download and install the latest binary distribution from the Comprehensive R Archive Network (CRAN). R can also be run from RStudio, which is a nice interface for working with R and provides a console. In order to use RStudio, R must be installed first. The latest version of RStudio is available here. R and RStudio should be accessible from the start menu, a desktop shortcut, or via command shell. Assuming you installed R on a 64-bit operating system (OS), two versions of R are made available, that is, a 32-bit and 64-bit version. The 64-bit version of R (R x64) is preferred.

R will require some packages in order to perform the tasks in the script. The packages can be installed from the console using the code below. The packages dataRetreival and EflowStats were developed by the USGS and provide the tools used in the script to retreive USGS data and compute streamflow statistics.

Site information

In order to produce the appopriate streamflow statistics, the date range that is available for the daily mean streamflow values (DVs) at each site is required. Additionally, time periods that may be considered “altered” and “unaltered” need to be evaluated. This script was developed to evaluate USGS peak-flow files in order to find periods of record in gages that may be affected by diversion, channelization, impoundment, and etc by looking at the range of available DVs and comparing that to codes in the peak-flow files.

Run EflowStats

# source a function written to apply the EflowStats package to a list of sites
source("https://raw.githubusercontent.com/bbreaker/RESTORE/master/R/runEflowStats.R")

# run in parallel
cl <- makePSOCKcluster(detectCores() - 4)

clusterEvalQ(cl, library(foreach)); registerDoParallel(cl)

newEflowDat <- foreach(i = 1:nrow(newSiteDF), .combine = dplyr::bind_rows, 
                       .packages = c("dplyr", "EflowStats", "dataRetrieval")) %dopar%
  (runEflowStats(site = newSiteDF[i,1], startDt = format(newSiteDF[i,2], "%Y-%m-%d"),
                 endDt = format(newSiteDF[i,3], "%Y-%m-%d")))

stopCluster(cl)

# get rid of NA rows produced
newEflowDat2 <- dplyr::filter(newEflowDat, comment %in% c("none", "EflowStats error", "EflowStats failure"))

# for sites that had multiple periods, get the weighted mean of the 
# periods using length of record as weight
EflowSites <- unique(newEflowDat2$site_no)

restoreEflows <- newEflowDat2[0,]

for(i in seq(1, length(EflowSites), 1)) {
  
  subEflowDat <- dplyr::filter(newEflowDat2, site_no == EflowSites[i])
  
  if(nrow(subEflowDat) == 1) {
    
    restoreEflows <- dplyr::bind_rows(restoreEflows, subEflowDat); rm(subEflowDat)
    
  }
  
  else if(nrow(subEflowDat) > 1) {
    
    subEflowDat$nYears <- as.numeric(stringr::str_sub(subEflowDat$end_date, 1, 4)) - 
      as.numeric(stringr::str_sub(subEflowDat$start_date, 1, 4))
    
    subEflowDatWghtd <- subEflowDat %>%
      summarize_at(c(4:181), funs(weighted.mean(., nYears, na.rm = TRUE))) %>%
      summarize_at(c(1:178), funs(round(., digits = 3)))
    
    info <- data.frame(site_no = EflowSites[i], start_date = subEflowDat[1,2],
                       end_date = subEflowDat[nrow(subEflowDat),3])
    
    subEflowDatNew <- dplyr::bind_cols(info, subEflowDatWghtd)
    
    subEflowDatNew$comment <- "weighted"
    
    restoreEflows <- dplyr::bind_rows(restoreEflows, subEflowDatNew); 
    rm(subEflowDat); rm(subEflowDatNew); rm(info)
    
  }
  
}

write.csv(restoreEflows, "data/restoreEflows.csv", row.names = FALSE)

restoreEflowsMap <- readr::read_csv("data/restoreEflows.csv")

m <- leaflet::leaflet()

grp <- c("USGS Topo", "USGS Imagery Only", "USGS Imagery Topo",
         "USGS Shaded Relief", "Hydrography")

att <- paste0("<a href='https://www.usgs.gov/'>",
              "U.S. Geological Survey</a> | ",
              "<a href='https://www.usgs.gov/laws/policies_notices.html'>",
              "Policies</a>")

GetURL <- function(service, host = "basemap.nationalmap.gov") {
  sprintf("https://%s/arcgis/services/%s/MapServer/WmsServer", host, service)
}

m <- leaflet::addWMSTiles(m, GetURL("USGSTopo"),
                            group = grp[1], attribution = att, layers = "0")
m <- leaflet::addWMSTiles(m, GetURL("USGSImageryOnly"),
                            group = grp[2], attribution = att, layers = "0")
m <- leaflet::addWMSTiles(m, GetURL("USGSImageryTopo"),
                            group = grp[3], attribution = att, layers = "0")
m <- leaflet::addWMSTiles(m, GetURL("USGSShadedReliefOnly"),
                            group = grp[4], attribution = att, layers = "0")

opt <- leaflet::WMSTileOptions(format = "image/png", transparent = TRUE)
m <- leaflet::addWMSTiles(m, GetURL("USGSHydroCached"),
                            group = grp[5], options = opt, layers = "0")
m <- leaflet::hideGroup(m, grp[5])

sites <- dataRetrieval::readNWISsite(restoreEflowsMap[1,1])

sites$project_no <- as.character(sites$project_no)

sites$construction_dt <- as.character(sites$construction_dt)

for(i in seq(1, nrow(restoreEflowsMap), 1)) {
  
  sitesNew <- dataRetrieval::readNWISsite(restoreEflowsMap[i,1])
  
  sitesNew$project_no <- as.character(sitesNew$project_no)
  
  sitesNew$construction_dt <- as.character(sitesNew$construction_dt)
  
  sites <- dplyr::bind_rows(sites, sitesNew); rm(sitesNew)
  
}

sp::coordinates(sites) <- c("dec_long_va", "dec_lat_va")

sp::proj4string(sites) <- sp::CRS("+proj=longlat +datum=NAD83")

sites <- sp::spTransform(sites, sp::CRS("+init=epsg:4326"))

num <- sites$site_no  # site number
nam <- sites$station_nm # local site name
url <- sprintf("https://waterdata.usgs.gov/nwis/inventory/?site_no=%s", num)
pop <- sprintf("<b>Name:</b> %s<br/><b>Site No:</b> <a href='%s'>%s</a>",
               nam, url, num)

opt <- leaflet::markerClusterOptions(showCoverageOnHover = TRUE)

m <- leaflet::addCircleMarkers(m, radius = 10, weight = 3, popup = pop,
                                 clusterOptions = opt, data = sites)

opt <- leaflet::layersControlOptions(collapsed = TRUE)

m <- leaflet::addLayersControl(m, baseGroups = grp[1:4],
                                 overlayGroups = grp[5], options = opt)
